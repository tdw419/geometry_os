   def cmd_snapshot_create(args):
       """Create snapshot of running container."""
       from systems.pixel_compiler.boot.multi_boot_manager import MultiBootManager
       from systems.pixel_compiler.boot.vm_snapshot import SnapshotError

       manager = MultiBootManager()
       container = manager.get_container(args.container)

       if not container:
           print(f"Error: Container '{args.container}' not found", file=sys.stderr)
           return 1

       if args.verbose:
           print(f"Creating snapshot '{args.tag}' for container '{args.container}'...")

       try:
           result = manager.create_snapshot(args.container, args.tag, description=args.description or "")

           if result.success:
               print(f"Snapshot '{args.tag}' created successfully")
               if args.verbose and result.metadata:
                   print(f"  ID: {result.metadata.snapshot_id}")
                   print(f"  State: {result.metadata.state.value}")
               return 0
           else:
               print(f"Failed to create snapshot: {result.error_message}", file=sys.stderr)
               return 1

       except SnapshotError as e:
           print(f"Error: {e}", file=sys.stderr)
           return 1
   ```

   def cmd_snapshot_list(args):
       """List snapshots for a container."""
       from systems.pixel_compiler.boot.multi_boot_manager import MultiBootManager
       from systems.pixel_compiler.boot.vm_snapshot import SnapshotError

       manager = MultiBootManager()
       container = manager.get_container(args.container)

       if not container:
           print(f"Error: Container '{args.container}' not found", file=sys.stderr)
           return 1

       try:
           snapshots = manager.list_container_snapshots(args.container)

           if not snapshots:
               print(f"No snapshots found for container '{args.container}'")
               return 0

           print(f"Snapshots for {args.container}:")
           print("-" * 60)
           for snap in snapshots:
               print(f"  {snap.tag}")
               print(f"    ID: {snap.id}")
               print(f"    Size: {snap.size}")
               print(f"    Date: {snap.date}")
               print(f"    VM Clock: {snap.vm_clock}")
               print()

           return 0

       except SnapshotError as e:
           print(f"Error: {e}", file=sys.stderr)
           return 1
   ```

       """Delete snapshot from container."""
       from systems.pixel_compiler.boot.multi_boot_manager import MultiBootManager
       from systems.pixel_compiler.boot.vm_snapshot import SnapshotError

       manager = MultiBootManager()

       if not args.force:
           confirm = input(f"Delete snapshot '{args.tag}' from '{args.container}'? [y/N] ")
           if confirm.lower() != 'y':
               print("Cancelled")
               return 0

       try:
           result = manager.delete_snapshot(args.container, args.tag)

           if result.success:
               print(f"Snapshot '{args.tag}' deleted")
               return 0
           else:
               print(f"Failed to delete snapshot: {result.error_message}", file=sys.stderr)
               return 1

       except SnapshotError as e:
           print(f"Error: {e}", file=sys.stderr)
           return 1
