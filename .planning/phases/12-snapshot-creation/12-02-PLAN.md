3. **Add snapshot methods to BootBridge class:**
   ```python
   def create_snapshot(self, tag: str, description: str = "") -> SnapshotResult:
       """
       Create VM snapshot using QEMU savevm command.

       Args:
           tag: Snapshot identifier (alphanumeric, no spaces)
           description: Optional description for the snapshot

       Returns:
           SnapshotResult with success status and metadata

       Raises:
           SnapshotError: If VM is not running or snapshot fails
       """
       if not self._booted or self._qemu is None:
           raise SnapshotError("VM is not booted")

       manager = self._snapshot_manager
       if manager is None:
           raise SnapshotError("Snapshot manager not available")

       return manager.create_snapshot(tag, description)

   def list_snapshots(self) -> List[SnapshotInfo]:
       """
       List available VM snapshots.

       Returns:
           List of SnapshotInfo objects

       Raises:
           SnapshotError: If VM is not running
       """
       if not self._booted or self._qemu is None:
           raise SnapshotError("VM is not booted")

       manager = self._snapshot_manager
       if manager is None:
           raise SnapshotError("Snapshot manager not available")

       return manager.list_snapshots()

   def restore_snapshot(self, tag: str) -> SnapshotResult:
       """
       Restore VM from snapshot using QEMU loadvm command.

       Args:
           tag: Snapshot tag to restore

       Returns:
           SnapshotResult with success status

       Raises:
           SnapshotError: If VM is not running or restore fails
       """
       if not self._booted or self._qemu is None:
           raise SnapshotError("VM is not booted")

       manager = self._snapshot_manager
       if manager is None:
           raise SnapshotError("Snapshot manager not available")

       return manager.restore_snapshot(tag)

   def delete_snapshot(self, tag: str) -> SnapshotResult:
       """
       Delete VM snapshot using QEMU delvm command.

       Args:
           tag: Snapshot tag to delete

       Returns:
           SnapshotResult with success status

       Raises:
           SnapshotError: If VM is not running or delete fails
       """
       if not self._booted or self._qemu is None:
           raise SnapshotError("VM is not booted")

       manager = self._snapshot_manager
       if manager is None:
           raise SnapshotError("Snapshot manager not available")

       return manager.delete_snapshot(tag)
   ```
