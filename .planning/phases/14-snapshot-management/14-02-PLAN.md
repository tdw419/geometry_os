def list_container_snapshots(self, name: str) -> List[SnapshotInfo]:
    """
    List snapshots for a container.

    If VM is running, returns live snapshot info from QEMU.
    If VM is not running, returns stored metadata from SnapshotStorage.

    Args:
        name: Container name

    Returns:
        List of SnapshotInfo objects

    Raises:
        ValueError: If container doesn't exist
    """
    info = self._containers.get(name)
    if not info:
        raise ValueError(f"Container '{name}' does not exist")

    # Try to get live snapshots if VM is running
    if info.state == ContainerState.RUNNING:
        snapshot_manager = self._get_snapshot_manager(name)
        if snapshot_manager:
            vm_snapshots = snapshot_manager.list_snapshots()

            # Update stored metadata with live info
            for snap in vm_snapshots:
                from .snapshot_storage import SnapshotMetadata
                stored = SnapshotMetadata.from_snapshot_info(snap, name)
                self._snapshot_storage.save_metadata(name, stored)

            return vm_snapshots

    # Fallback to stored metadata
    stored_snapshots = self._snapshot_storage.list_snapshots(name)

    # Convert SnapshotMetadata to SnapshotInfo format
    result = []
    for stored in stored_snapshots:
        # Create SnapshotInfo-compatible object
        snap_info = SnapshotInfo(
            id=0,  # Not available from storage
            tag=stored.tag,
            size=stored.size,
            date=stored.created_at,
            vm_clock=stored.vm_clock or "00:00:00.000"
        )
        result.append(snap_info)

    return result
```

This ensures:
1. Live VM snapshots are preferred when VM is running
2. Metadata is updated with live info when available
3. Stored metadata is used as fallback when VM is stopped

def delete_snapshot(self, name: str, tag: str) -> SnapshotResult:
    """
    Delete a snapshot from a container.

    Args:
        name: Container name
        tag: Snapshot tag to delete

    Returns:
        SnapshotResult with success status

    Raises:
        ValueError: If container doesn't exist
    """
    info = self._containers.get(name)
    if not info:
        raise ValueError(f"Container '{name}' does not exist")

    result = SnapshotResult(success=False, tag=tag)

    # Try to delete from VM if running
    if info.state == ContainerState.RUNNING:
        snapshot_manager = self._get_snapshot_manager(name)
        if snapshot_manager:
            result = snapshot_manager.delete_snapshot(tag)

    # Always delete from storage (even if VM deletion failed or VM not running)
    storage_deleted = self._snapshot_storage.delete_metadata(name, tag)

    # Update ContainerInfo
    if name in self._containers:
        info = self._containers[name]
        if tag in info.snapshots:
            info.snapshots.remove(tag)

    # Consider success if either VM deletion or storage deletion succeeded
    if not result.success and storage_deleted:
        result = SnapshotResult(
            success=True,
            tag=tag,
            error_message=None
        )

    return result
```
