   async def delta_handler(path: str, method: str, headers: dict) -> Tuple[int, dict, bytes]:
       """
       Handle delta manifest requests.

       Routes:
           GET /delta/<container>.json -> return manifest
           GET /delta/list -> return list of available deltas
       """
       if path.startswith('/delta/list'):
           # Return list of available delta manifests
           return 200, {'Content-Type': 'application/json'}, json.dumps(deltas).encode()

       if path.startswith('/delta/') and path.endswith('.json'):
           # Return specific manifest
           manifest_name = path[7:]  # Strip /delta/
           manifest_path = http_root / 'delta' / manifest_name
           if manifest_path.exists():
               return 200, {'Content-Type': 'application/json'}, manifest_path.read_bytes()
           return 404, {'Content-Type': 'text/plain'}, b'Manifest not found'

       return 404, {'Content-Type': 'text/plain'}, b'Not found'

1. Add cmd_update(args) function:
   ```python
   def cmd_update(args):
       """Handle delta update command."""
       import urllib.request
       import json
       from systems.pixel_compiler.delta_manifest import DeltaManifest
       from systems.pixel_compiler.delta_patch import DeltaPatcher, HTTPByteFetcher

       server_url = args.server
       local_file = args.file
       output_file = args.output or local_file

       # 1. Fetch manifest from server
       manifest_url = f"{server_url}/delta/{Path(local_file).name}.json"
       print(f"Fetching manifest from {manifest_url}...")

       try:
           with urllib.request.urlopen(manifest_url) as response:
               manifest_json = response.read().decode('utf-8')
               manifest = DeltaManifest.from_json(manifest_json)
       except urllib.error.HTTPError as e:
           if e.code == 404:
               print("No delta available. File is up to date or no newer version exists.")
               return
           raise

       # 2. Check if update needed
       local_checksum = compute_file_checksum(local_file)
       if local_checksum == manifest.new_checksum:
           print("Already up to date.")
           return

       if local_checksum != manifest.old_checksum:
           print(f"Warning: Local file checksum mismatch.")
           print(f"  Expected: {manifest.old_checksum}")
           print(f"  Got: {local_checksum}")
           if not args.force:
               print("Use --force to proceed anyway.")
               return

       # 3. Fetch changed regions via HTTP Range requests
       fetcher = HTTPByteFetcher(server_url, Path(local_file).name)

       # 4. Apply patch
       patcher = DeltaPatcher(validate_checksums=not args.skip_validation)
       try:
           result = patcher.apply_with_fetcher(
               base_path=local_file,
               manifest=manifest,
               byte_fetcher=fetcher,
               output_path=output_file
           )
           print(f"Updated: {result}")
           print(f"Transferred: {fetcher.bytes_transferred:,} bytes")
           print(f"Saved: {manifest.new_size - fetcher.bytes_transferred:,} bytes")
       except PatchError as e:
           print(f"Patch failed: {e.message}", file=sys.stderr)
           sys.exit(1)
   ```

2. Add HTTPByteFetcher class to delta_patch.py (or inline):
   ```python
   class HTTPByteFetcher:
       """Fetches bytes via HTTP Range requests."""

       def __init__(self, base_url: str, filename: str):
           self.base_url = base_url.rstrip('/')
           self.filename = filename
           self.bytes_transferred = 0

       def fetch_region(self, offset: int, length: int) -> bytes:
           url = f"{self.base_url}/{self.filename}"
           headers = {'Range': f'bytes={offset}-{offset + length - 1}'}

           req = urllib.request.Request(url, headers=headers)
           with urllib.request.urlopen(req) as response:
               data = response.read()
               self.bytes_transferred += len(data)
               return data
   ```
