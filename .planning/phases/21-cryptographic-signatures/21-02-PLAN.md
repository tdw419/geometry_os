```python
from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PrivateKey, Ed25519PublicKey
from cryptography.hazmat.primitives import serialization
from cryptography.exceptions import InvalidSignature
from pathlib import Path
from typing import Optional, Tuple
from PIL import Image
from PIL.PngImagePlugin import PngInfo
from io import BytesIO
import json

class FileSigner:
    """
    Signs PixelRTS files with Ed25519 signatures.

    Usage:
        # Generate keypair
        private_key, public_key_bytes = FileSigner.generate_keypair()

        # Save private key
        FileSigner.save_private_key(private_key, Path("private.pem"))

        # Sign a file
        signer = FileSigner(private_key)
        signer.sign_file(Path("boot.rts.png"))
    """

    def __init__(self, private_key: Ed25519PrivateKey):
        self.private_key = private_key
        self.public_key = private_key.public_key()

    def sign_file(self, png_path: Path, output_path: Optional[Path] = None) -> Path:
        """
        Sign a .rts.png file and add signature to metadata.

        Args:
            png_path: Path to the file to sign
            output_path: Optional output path (defaults to overwriting input)

        Returns:
            Path to the signed file

        Raises:
            ValueError: If file is not a valid PixelRTS file or has no data hash
        """
        output_path = output_path or png_path

        # Read PNG and extract metadata
        png_data = png_path.read_bytes()
        img = Image.open(BytesIO(png_data))

        # Find and parse PixelRTS metadata
        metadata = self._extract_metadata(img)
        if metadata is None:
            raise ValueError("Not a valid PixelRTS file - no metadata found")

        # Get data hash to sign (SHA256 hash already in metadata)
        data_hash_hex = metadata.get("hashes", {}).get("data")
        if data_hash_hex is None:
            raise ValueError("No data hash found in metadata - file must have consistency hash")

        data_hash = bytes.fromhex(data_hash_hex)

        # Sign the hash (not the full data)
        signature = self.private_key.sign(data_hash)

        # Add signature to metadata
        metadata["signature"] = {
            "algorithm": "Ed25519",
            "public_key": self.public_key.public_bytes_raw().hex(),
            "value": signature.hex()
        }

        # Update PNG with new metadata
        return self._write_metadata(img, metadata, output_path)

    def _extract_metadata(self, img: Image.Image) -> Optional[dict]:
        """Extract PixelRTS metadata from PNG."""
        for key, value in img.text.items():
            if "PixelRTS" in str(value):
                try:
                    from systems.pixel_compiler.pixelrts_v2_core import PixelRTSMetadata
                    return PixelRTSMetadata.decode_png_text(value.encode("utf-8"))
                except (ValueError, json.JSONDecodeError):
                    continue
        return None

    def _write_metadata(self, img: Image.Image, metadata: dict, output_path: Path) -> Path:
        """Write updated metadata to PNG file."""
        # Create new PNG with metadata
        pnginfo = PngInfo()

        # Encode metadata
        metadata_str = "PixelRTS" + json.dumps(metadata, separators=(",", ":"))
        pnginfo.add_text("PixelRTS", metadata_str)

        # Preserve existing text chunks (except PixelRTS)
        for key, value in img.text.items():
            if key != "PixelRTS":
                pnginfo.add_text(key, value)

        # Save
        buffer = BytesIO()
        img.save(buffer, format="PNG", pnginfo=pnginfo)
        output_path.write_bytes(buffer.getvalue())

        return output_path

    @staticmethod
    def generate_keypair() -> Tuple[Ed25519PrivateKey, bytes]:
        """
        Generate a new Ed25519 keypair.

        Returns:
            Tuple of (private_key, public_key_bytes)
        """
        private_key = Ed25519PrivateKey.generate()
        public_key_bytes = private_key.public_key().public_bytes_raw()
        return private_key, public_key_bytes

    @staticmethod
    def save_private_key(private_key: Ed25519PrivateKey, path: Path) -> None:
        """
        Save private key to PEM file.

        Args:
            private_key: The private key to save
            path: Output path for PEM file
        """
        pem = private_key.private_bytes(
            encoding=serialization.Encoding.PEM,
            format=serialization.PrivateFormat.PKCS8,
            encryption_algorithm=serialization.NoEncryption()
        )
        path.write_bytes(pem)

    @staticmethod
    def load_private_key(path: Path) -> Ed25519PrivateKey:
        """
        Load private key from PEM file.

        Args:
            path: Path to PEM file

        Returns:
            Ed25519PrivateKey instance
        """
        pem_data = path.read_bytes()
        return serialization.load_pem_private_key(pem_data, password=None)
```

Key details:
- Sign the HASH (32 bytes), not the full data - matches SignatureVerifier expectation
- Use PKCS8 format for private key PEM (standard, interoperable)
- Preserve existing PNG text chunks when updating metadata
- Raise ValueError for invalid files (not PixelRTS, missing hash)
