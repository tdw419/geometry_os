```python
def cmd_ps(args):
    """Handle ps command - List running PixelRTS containers."""
    from systems.pixel_compiler.boot import ContainerState

    # Check for state file (JSON file tracking running containers)
    state_file = Path(args.state_file) if args.state_file else Path("/tmp/pixelrts/containers.json")

    if not state_file.exists():
        print("No containers running.")
        return 0

    import json
    try:
        with open(state_file) as f:
            containers = json.load(f)
    except (json.JSONDecodeError, IOError) as e:
        print(f"Error reading state file: {e}", file=sys.stderr)
        return 1

    if not containers:
        print("No containers running.")
        return 0

    # Format output
    if args.json:
        print(json.dumps(containers, indent=2))
    else:
        # Header
        print(f"{'NAME':<30} {'STATE':<12} {'VNC':<8} {'PID':<8}")
        print("-" * 60)

        for c in containers:
            name = c.get("name", "unknown")[:28]
            state = c.get("state", "unknown")
            vnc = c.get("vnc_port", "-")
            pid = c.get("pid", "-")
            print(f"{name:<30} {state:<12} {vnc:<8} {pid:<8}")

    return 0
```

```python
def _save_state(self) -> None:
    """Persist container state to file for pixelrts ps."""
    import json
    self._state_file.parent.mkdir(parents=True, exist_ok=True)

    containers = [
        info.to_dict() for info in self._containers.values()
    ]

    with open(self._state_file, "w") as f:
        json.dump(containers, f, indent=2)
```
