@dataclass
class AllocatedResources:
    """Resources allocated for a single container."""
    container_id: str
    vnc_port: int          # Actual port number (5900 + display)
    vnc_display: int       # VNC display number
    serial_socket: Path    # Serial console socket path
    monitor_socket: Path   # QEMU monitor socket path

class ResourceAllocator:
    """Allocates unique resources for multi-container boot."""

    VNC_PORT_BASE = 5900
    VNC_PORT_MAX = 5999    # Max 100 concurrent containers

    def __init__(self, socket_dir: Path | None = None):
        self.socket_dir = socket_dir or Path("/tmp/pixelrts")
        self._allocated_ports: set[int] = set()
        self._next_port = self.VNC_PORT_BASE
        self._allocated_ids: set[str] = set()

    def allocate(self, container_name: str) -> AllocatedResources:
        """Allocate unique resources for a container.

        Raises:
            ResourceExhaustedError: If no VNC ports available
        """
        # Find next available port
        while self._next_port in self._allocated_ports:
            self._next_port += 1
            if self._next_port > self.VNC_PORT_MAX:
                raise ResourceExhaustedError("No VNC ports available")

        port = self._next_port
        self._allocated_ports.add(port)
        self._next_port += 1

        # Generate unique ID from container name (deterministic)
        container_id = self._make_unique_id(container_name)

        return AllocatedResources(
            container_id=container_id,
            vnc_port=port,
            vnc_display=port - self.VNC_PORT_BASE,
            serial_socket=self.socket_dir / f"{container_id}-serial.sock",
            monitor_socket=self.socket_dir / f"{container_id}-monitor.sock",
        )

    def release(self, resources: AllocatedResources) -> None:
        """Release allocated resources for reuse."""
        self._allocated_ports.discard(resources.vnc_port)
        self._allocated_ids.discard(resources.container_id)
        # Clean up socket files if they exist
        for sock in [resources.serial_socket, resources.monitor_socket]:
            if sock.exists():
                sock.unlink()

    def _make_unique_id(self, name: str) -> str:
        """Generate unique ID from container name."""
        # Use MD5 hash prefix for uniqueness + readable suffix
        import hashlib
        base = name.replace(".rts.png", "").replace(".png", "")
        hash_prefix = hashlib.md5(name.encode()).hexdigest()[:6]
        unique_id = f"{base[:12]}-{hash_prefix}"
        # Handle collisions
        counter = 0
        while unique_id in self._allocated_ids:
            counter += 1
            unique_id = f"{base[:12]}-{hash_prefix}-{counter}"
        self._allocated_ids.add(unique_id)
        return unique_id
