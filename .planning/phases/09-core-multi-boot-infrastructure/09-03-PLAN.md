def _cleanup_successful_containers(self, containers: List[ContainerInfo]) -> None:
    """Stop all successfully started containers (compensating transaction)."""
    for info in containers:
        if info.state == ContainerState.RUNNING and info.bridge:
            logger.info(f"Cleaning up container: {info.name}")
            try:
                info.bridge.stop()
                if info.resources:
                    self._allocator.release(info.resources)
            except Exception as e:
                logger.warning(f"Error cleaning up {info.name}: {e}")
            info.state = ContainerState.STOPPED
```

3. Modify _boot_all_async to call cleanup when failures occur:

        )

def test_cleanup_on_partial_failure(self, tmp_path, monkeypatch):
    manager = MultiBootManager(socket_dir=tmp_path)

    # Create two test files
    file1 = tmp_path / "a.rts.png"
    file2 = tmp_path / "b.rts.png"
    file1.write_bytes(b"fake")
    file2.write_bytes(b"fake")

    call_count = 0
    def mock_boot(self, cmdline=None, extra_qemu_args=None):
        nonlocal call_count
        call_count += 1
        if call_count == 1:
            return BootResult(success=True, pid=100, vnc_port=5900)
        else:
            return BootResult(success=False, error_message="QEMU failed")

    monkeypatch.setattr(BootBridge, "boot", mock_boot)

    result = manager.boot_all([file1, file2], cleanup_on_failure=True)

    assert not result.success
    assert result.cleanup_performed
    assert result.containers[0].state == ContainerState.STOPPED
```
