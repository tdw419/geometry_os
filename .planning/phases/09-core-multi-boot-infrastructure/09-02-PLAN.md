```python
import asyncio
import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Dict, List, Optional
from enum import Enum

from .boot_bridge import BootBridge, BootResult
from .resource_allocator import ResourceAllocator, AllocatedResources

logger = logging.getLogger(__name__)


class ContainerState(Enum):
    """State of a container in the multi-boot system."""
    PENDING = "pending"
    LAUNCHING = "launching"
    RUNNING = "running"
    STOPPED = "stopped"
    FAILED = "failed"


@dataclass
class ContainerInfo:
    """Information about a managed container."""
    name: str
    path: Path
    state: ContainerState
    resources: Optional[AllocatedResources] = None
    boot_result: Optional[BootResult] = None
    bridge: Optional[BootBridge] = None
    error_message: Optional[str] = None

    def to_dict(self) -> dict:
        return {
            "name": self.name,
            "path": str(self.path),
            "state": self.state.value,
            "vnc_port": self.resources.vnc_port if self.resources else None,
            "pid": self.boot_result.pid if self.boot_result else None,
            "error": self.error_message,
        }


@dataclass
class MultiBootResult:
    """Result of a multi-boot operation."""
    success: bool
    containers: List[ContainerInfo]
    failed_count: int
    error_messages: List[str]


class MultiBootManager:
    """Manages concurrent boot of multiple PixelRTS containers.

    Uses asyncio for concurrent QEMU startup while ensuring unique
    resource allocation per container.
    """

    def __init__(
        self,
        memory: str = "2G",
        cpus: int = 2,
        verbose: bool = False,
        socket_dir: Optional[Path] = None,
    ):
        self.memory = memory
        self.cpus = cpus
        self.verbose = verbose
        self._allocator = ResourceAllocator(socket_dir=socket_dir)
        self._containers: Dict[str, ContainerInfo] = {}

    def boot_all(
        self,
        paths: List[Path],
        cmdline: Optional[str] = None,
        stagger_delay: float = 0.5,
    ) -> MultiBootResult:
        """Boot multiple containers concurrently.

        Args:
            paths: List of .rts.png file paths to boot
            cmdline: Optional kernel command line for all containers
            stagger_delay: Seconds to wait between container launches
                          (prevents resource contention/OOM)

        Returns:
            MultiBootResult with status of all containers
        """
        # Validate all paths exist first
        for path in paths:
            if not path.exists():
                return MultiBootResult(
                    success=False,
                    containers=[],
                    failed_count=len(paths),
                    error_messages=[f"File not found: {path}"],
                )

        # Create container info for each
        for path in paths:
            name = path.name
            self._containers[name] = ContainerInfo(
                name=name,
                path=path,
                state=ContainerState.PENDING,
            )

        # Run async boot
        return asyncio.run(self._boot_all_async(paths, cmdline, stagger_delay))

    async def _boot_all_async(
        self,
        paths: List[Path],
        cmdline: Optional[str],
        stagger_delay: float,
    ) -> MultiBootResult:
        """Async implementation of boot_all."""
        tasks = []
        for i, path in enumerate(paths):
            # Stagger starts to avoid resource contention
            if i > 0:
                await asyncio.sleep(stagger_delay)
            tasks.append(self._boot_single(path, cmdline))

        # Wait for all boots to complete
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # Process results
        containers = []
        failed_count = 0
        errors = []

        for path, result in zip(paths, results):
            name = path.name
            info = self._containers[name]

            if isinstance(result, Exception):
                info.state = ContainerState.FAILED
                info.error_message = str(result)
                failed_count += 1
                errors.append(f"{name}: {result}")
            elif result.success:
                info.state = ContainerState.RUNNING
                info.boot_result = result
            else:
                info.state = ContainerState.FAILED
                info.error_message = result.error_message
                failed_count += 1
                errors.append(f"{name}: {result.error_message}")

            containers.append(info)

        return MultiBootResult(
            success=(failed_count == 0),
            containers=containers,
            failed_count=failed_count,
            error_messages=errors,
        )

    async def _boot_single(
        self,
        path: Path,
        cmdline: Optional[str],
    ) -> BootResult:
        """Boot a single container (async wrapper)."""
        name = path.name
        info = self._containers[name]

        # Allocate resources
        info.state = ContainerState.LAUNCHING
        info.resources = self._allocator.allocate(name)

        # Create BootBridge with allocated resources
        bridge = BootBridge(
            rts_png_path=path,
            memory=self.memory,
            cpus=self.cpus,
            vnc_display=info.resources.vnc_display,
            verbose=self.verbose,
        )
        info.bridge = bridge

        # Boot (run in executor to avoid blocking)
        loop = asyncio.get_event_loop()
        result = await loop.run_in_executor(
            None,
            lambda: bridge.boot(cmdline=cmdline),
        )

        return result

    def list_containers(self) -> List[ContainerInfo]:
        """List all managed containers and their states."""
        return list(self._containers.values())

    def get_container(self, name: str) -> Optional[ContainerInfo]:
        """Get info for a specific container."""
        return self._containers.get(name)

    def stop(self, name: str) -> bool:
        """Stop a specific container."""
        info = self._containers.get(name)
        if info and info.bridge:
            info.bridge.stop()
            if info.resources:
                self._allocator.release(info.resources)
            info.state = ContainerState.STOPPED
            return True
        return False

    def stop_all(self) -> None:
        """Stop all managed containers."""
        for name in list(self._containers.keys()):
            self.stop(name)

    def cleanup(self) -> None:
        """Remove stopped/failed containers from tracking."""
        to_remove = [
            name for name, info in self._containers.items()
            if info.state in (ContainerState.STOPPED, ContainerState.FAILED)
        ]
        for name in to_remove:
            del self._containers[name]
```

Key design decisions:
- stagger_delay prevents OOM from simultaneous QEMU startup
- asyncio.run_in_executor keeps BootBridge sync (it uses subprocess.Popen)
- Resource allocation happens per-container before BootBridge creation
