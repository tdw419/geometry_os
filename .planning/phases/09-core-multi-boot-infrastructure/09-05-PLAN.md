```python
def cmd_boot(args):
    """Handle boot command - Boot .rts.png files with QEMU."""
    from systems.pixel_compiler.boot import BootBridge, BootResult

    # Multiple files: use MultiBootManager

def _boot_multiple(args, input_paths: list[Path]) -> int:
    """Boot multiple containers using MultiBootManager."""
    # Validate all files exist first
    for path in input_paths:
        if not path.exists():
            print(f"Error: Input file not found: {path}", file=sys.stderr)
            return 1

    if args.verbose:
        print(f"Booting {len(input_paths)} containers...")
        for p in input_paths:
            print(f"  - {p}")

    try:
        manager = MultiBootManager(
            memory=args.memory,
            cpus=args.cpus,
            verbose=not args.quiet and (args.verbose or sys.stdout.isatty()),
        )

        result = manager.boot_all(
            paths=input_paths,
            cmdline=args.cmdline,
            stagger_delay=0.5,  # 500ms between container starts
        )

        # Print results
        if not args.quiet:
            print(f"\nBoot {'successful' if result.success else 'completed with errors'}!")
            print(f"  Running: {len(input_paths) - result.failed_count}/{len(input_paths)}")

            for info in result.containers:
                if info.state == ContainerState.RUNNING:
                    print(f"  - {info.name}: VNC :{info.resources.vnc_display} (port {info.resources.vnc_port})")
                else:
                    print(f"  - {info.name}: FAILED - {info.error_message}")

        # Handle background mode
        if args.background:
            if not args.quiet:
                print(f"\nRunning in background. Use 'pixelrts ps' to check status.")
            return 0 if result.success else 1

        # Wait for containers (Ctrl+C to stop all)
        if not args.quiet:
            print("\nContainers running. Press Ctrl+C to stop all...")

        # Set up signal handler for graceful shutdown
        shutdown_requested = False

        def handle_shutdown(signum, frame):
            nonlocal shutdown_requested
            shutdown_requested = True
            if not args.quiet:
                print("\nStopping all containers...")

        original_sigint = signal.signal(signal.SIGINT, handle_shutdown)
        original_sigterm = signal.signal(signal.SIGTERM, handle_shutdown)

        try:
            # Wait for signal
            while not shutdown_requested:
                signal.pause()
        finally:
            signal.signal(signal.SIGINT, original_sigint)
            signal.signal(signal.SIGTERM, original_sigterm)
            manager.stop_all()

        return 0 if result.success else 1

    except ImportError as e:
        print(f"Error: Required module not available: {e}", file=sys.stderr)
        return 1
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        if args.verbose:
            import traceback
            traceback.print_exc()
        return 1
```
