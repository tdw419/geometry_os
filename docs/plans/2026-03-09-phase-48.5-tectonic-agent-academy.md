```python
    # Counter-bid strategy

    counter_bid_threshold: float = 0.3   # only counter if < 30% of budget

    # Timing strategy

    counter_timing_ms: float = 400.0  # counter-bid late for surprise

    # Risk tolerance
    aggression_level: float = 0.5   # 0=conservative, 1=aggressive
    budget_reserve_fraction: float = 0.2  # keep 20% in reserve
```

### Agent Strategy Types (for training diversity)

| Strategy | Description | Bid Behavior |
|----------|-------------|--------------|
| `random` | Baseline | Random bid 0.1-0.9 of budget |
| `aggressive` | High risk | Bids 80% of budget frequently |
| `conservative` | Low risk | Bids 20% of budget, rare counters |
| `distance_aware` | Smart | Bids inversely proportional to distance |
| `adaptive` | Learning | Adjusts based on recent outcomes |

```python
# systems/tectonic/strategy.py
"""
Tectonic Strategy Profile - Learned parameters for optimal bidding.

Phase 48.5: Tectonic Agent Academy
"""

import math
from dataclasses import dataclass, field
from enum import Enum
from typing import Dict, Any, Tuple, Optional


class StrategyType(Enum):
    """Preset strategy types for training diversity."""
    RANDOM = "random"
    AGGRESSIVE = "aggressive"
    CONSERVATIVE = "conservative"
    DISTANCE_AWARE = "distance_aware"
    ADAPTIVE = "adaptive"


@dataclass
class StrategyProfile:
    """
    Learned strategy parameters for optimal bidding.

    Controls how a TectonicAgent calculates bids and counter-bids
    based on tile distance, budget, and learned risk tolerance.
    """

    # Base bid calculation
    base_bid_fraction: float = 0.1
    distance_weight: float = 0.05
    distance_decay: float = 0.5

    # Counter-bid strategy
    counter_bid_multiplier: float = 1.1
    counter_bid_threshold: float = 0.3

    # Timing strategy
    bid_timing_ms: float = 250.0
    counter_timing_ms: float = 400.0

    # Risk tolerance
    aggression_level: float = 0.5
    budget_reserve_fraction: float = 0.2

    def calculate_optimal_bid(
        self,
        tile: Tuple[int, int],
        purpose: str,
        budget: float
    ) -> float:
        """
        Calculate optimal bid for a tile claim.

        Formula:
            base = budget * base_bid_fraction
            distance_factor = 1 / (1 + distance * distance_weight)
            bid = base * distance_factor * (1 + aggression_level * 0.5)

        Args:
            tile: (x, y) coordinates
            purpose: Purpose string (may affect future weights)
            budget: Available budget

        Returns:
            Optimal bid amount
        """
        # Calculate distance from origin
        x, y = tile
        distance = math.sqrt(x * x + y * y)

        # Base bid
        base = budget * self.base_bid_fraction

        # Distance factor (closer = higher bid)
        distance_factor = 1.0 / (1.0 + distance * self.distance_weight)

        # Aggression modifier
        aggression_mod = 1.0 + self.aggression_level * 0.5

        # Calculate final bid
        bid = base * distance_factor * aggression_mod

        # Ensure we don't exceed available budget (minus reserve)
        max_bid = budget * (1.0 - self.budget_reserve_fraction)
        return min(bid, max_bid)

    def should_counter_bid(
        self,
        original_bid: float,
        my_budget: float,
        tile: Optional[Tuple[int, int]] = None
    ) -> bool:
        """
        Decide whether to counter-bid on a claim.

        Args:
            original_bid: The original claim's bid
            my_budget: Our available budget
            tile: Optional tile for distance consideration

        Returns:
            True if we should counter-bid
        """
        # Don't counter if original bid is too high
        if original_bid > my_budget * self.counter_bid_threshold:
            return False

        # Counter-bid probability based on aggression
        counter_prob = self.aggression_level * 0.7

        # Simple threshold check for deterministic tests
        return original_bid < my_budget * self.counter_bid_threshold

    def calculate_counter_bid(self, original_bid: float) -> float:
        """
        Calculate counter-bid amount.

        Args:
            original_bid: The original claim's bid

        Returns:
            Counter-bid amount
        """
        return original_bid * self.counter_bid_multiplier

    def bid_timing_preference(self) -> float:
        """Get preferred timing for initial bids (ms into 500ms window)."""
        return self.bid_timing_ms

    def counter_timing_preference(self) -> float:
        """Get preferred timing for counter-bids (ms into 500ms window)."""
        return self.counter_timing_ms

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to dictionary."""
        return {
            "base_bid_fraction": self.base_bid_fraction,
            "distance_weight": self.distance_weight,
            "distance_decay": self.distance_decay,
            "counter_bid_multiplier": self.counter_bid_multiplier,
            "counter_bid_threshold": self.counter_bid_threshold,
            "bid_timing_ms": self.bid_timing_ms,
            "counter_timing_ms": self.counter_timing_ms,
            "aggression_level": self.aggression_level,
            "budget_reserve_fraction": self.budget_reserve_fraction,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "StrategyProfile":
        """Deserialize from dictionary."""
        return cls(**data)

    @classmethod
    def get_preset(cls, strategy_type: StrategyType) -> "StrategyProfile":
        """
        Get a preset strategy profile by type.

        Args:
            strategy_type: The type of preset strategy

        Returns:
            StrategyProfile configured for that strategy type
        """
        presets = {
            StrategyType.RANDOM: cls(
                base_bid_fraction=0.3,
                aggression_level=0.5,
                distance_weight=0.0,  # Ignore distance
            ),
            StrategyType.AGGRESSIVE: cls(
                base_bid_fraction=0.25,
                distance_weight=0.02,
                counter_bid_multiplier=1.3,
                counter_bid_threshold=0.5,
                aggression_level=0.9,
                budget_reserve_fraction=0.1,
            ),
            StrategyType.CONSERVATIVE: cls(
                base_bid_fraction=0.08,
                distance_weight=0.08,
                counter_bid_multiplier=1.05,
                counter_bid_threshold=0.15,
                aggression_level=0.2,
                budget_reserve_fraction=0.4,
            ),
            StrategyType.DISTANCE_AWARE: cls(
                base_bid_fraction=0.12,
                distance_weight=0.1,
                distance_decay=0.3,
                counter_bid_multiplier=1.15,
                aggression_level=0.5,
            ),
            StrategyType.ADAPTIVE: cls(
                base_bid_fraction=0.15,
                distance_weight=0.05,
                aggression_level=0.5,
                budget_reserve_fraction=0.25,
            ),
        }
        return presets.get(strategy_type, cls())
```

```python
# systems/tectonic/arena.py
"""
Simulation Arena - Run mock negotiations for strategy training.

Phase 48.5: Tectonic Agent Academy

The arena runs thousands of simulated negotiations between agents
with different strategies to collect win/loss data for learning.
"""

import asyncio
import logging
import random
import time
from dataclasses import dataclass, field
from typing import Dict, List, Any, Tuple, Optional

from systems.tectonic.claim import TectonicClaim, TectonicBid, ClaimStatus
from systems.tectonic.negotiator import TectonicNegotiator
from systems.tectonic.agent import TectonicAgent
from systems.tectonic.strategy import StrategyProfile, StrategyType

logger = logging.getLogger("tectonic.arena")


@dataclass
class AgentConfig:
    """Configuration for a simulated agent."""

    agent_id: str
    budget: float = 100.0
    strategy_type: StrategyType = StrategyType.RANDOM


@dataclass
class SimulationResult:
    """Result of a single negotiation round."""

    round_id: int
    winner: str
    tile: Tuple[int, int]
    winning_bid: float
    participants: List[str]
    forces: Dict[str, float]
    timestamp: float = field(default_factory=time.time)

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to dictionary."""
        return {
            "round_id": self.round_id,
            "winner": self.winner,
            "tile": self.tile,
            "winning_bid": self.winning_bid,
            "participants": self.participants,
            "forces": self.forces,
            "timestamp": self.timestamp,
        }


class SimulatedAgent:
    """Agent that uses a strategy profile for decisions."""

    def __init__(self, config: AgentConfig):
        self.config = config
        self.agent_id = config.agent_id
        self.budget = config.budget
        self.strategy = StrategyProfile.get_preset(config.strategy_type)
        self.wins = 0
        self.losses = 0
        self.budget_consumed = 0.0

    def calculate_bid(self, tile: Tuple[int, int], purpose: str) -> float:
        """Calculate bid based on strategy."""
        bid = self.strategy.calculate_optimal_bid(tile, purpose, self.budget)
        # Add some randomness for RANDOM strategy
        if self.config.strategy_type == StrategyType.RANDOM:
            bid = random.uniform(self.budget * 0.1, self.budget * 0.9)
        return bid

    def should_counter(self, original_bid: float) -> bool:
        """Decide whether to counter-bid."""
        return self.strategy.should_counter_bid(original_bid, self.budget)

    def calculate_counter(self, original_bid: float) -> float:
        """Calculate counter-bid amount."""
        return self.strategy.calculate_counter_bid(original_bid)


class SimulationArena:
    """
    Arena for running simulated tectonic negotiations.

    Creates agents with different strategies, runs negotiation rounds,
    and collects statistics for strategy learning.
    """

    def __init__(self, agent_configs: Optional[List[AgentConfig]] = None):
        """Initialize the arena."""
        self.agent_configs = agent_configs or []
        self.results: List[SimulationResult] = []
        self.rounds_completed = 0
        self._win_counts: Dict[str, int] = {}
        self._total_budget_consumed = 0.0

    def add_agent(self, config: AgentConfig) -> None:
        """Add an agent configuration."""
        self.agent_configs.append(config)

    def run_round(
        self,
        tile: Optional[Tuple[int, int]] = None,
        purpose: str = "training"
    ) -> SimulationResult:
        """
        Run a single negotiation round.

        Args:
            tile: Optional specific tile (random if not provided)
            purpose: Purpose string for the claim

        Returns:
            SimulationResult with winner and details
        """
        if tile is None:
            # Random tile
            tile = (random.randint(0, 100), random.randint(0, 100))

        # Create fresh negotiator and agents for this round
        negotiator = TectonicNegotiator()
        agents: List[SimulatedAgent] = []

        for config in self.agent_configs:
            agents.append(SimulatedAgent(config))

        if not agents:
            raise ValueError("No agents configured")

        # First agent makes claim
        claimant = agents[0]
        bid = claimant.calculate_bid(tile, purpose)

        claim = TectonicClaim(
            tile=tile,
            purpose=purpose,
            bid=bid,
            agent_id=claimant.agent_id,
        )
        claim_id = negotiator.submit_claim(claim)

        # Other agents may counter-bid
        for agent in agents[1:]:
            if agent.should_counter(bid):
                counter = agent.calculate_counter(bid)
                counter_bid = TectonicBid(
                    claim_id=claim_id,
                    amount=counter,
                    agent_id=agent.agent_id,
                )
                negotiator.submit_bid(counter_bid)

        # Settle
        settlement = negotiator.settle(claim_id)

        # Record result
        self.rounds_completed += 1
        result = SimulationResult(
            round_id=self.rounds_completed,
            winner=settlement.winner,
            tile=tile,
            winning_bid=settlement.claim.bid if settlement.winner == claimant.agent_id else max(b.amount for b in negotiator.get_bids_for_claim(claim_id)) if negotiator.get_bids_for_claim(claim_id) else settlement.claim.bid,
            participants=[a.agent_id for a in agents],
            forces={a.agent_id: 0.0 for a in agents},  # Simplified
        )

        self.results.append(result)
        self._win_counts[result.winner] = self._win_counts.get(result.winner, 0) + 1

        return result

    def run_rounds(
        self,
        count: int,
        tiles: Optional[List[Tuple[int, int]]] = None
    ) -> List[SimulationResult]:
        """
        Run multiple negotiation rounds.

        Args:
            count: Number of rounds to run
            tiles: Optional list of specific tiles

        Returns:
            List of SimulationResults
        """
        results = []
        for i in range(count):
            tile = tiles[i] if tiles and i < len(tiles) else None
            result = self.run_round(tile=tile)
            results.append(result)
        return results

    async def run_rounds_async(
        self,
        count: int,
        batch_size: int = 10
    ) -> List[SimulationResult]:
        """
        Run rounds asynchronously in batches.

        Args:
            count: Total rounds to run
            batch_size: Rounds per batch

        Returns:
            List of all SimulationResults
        """
        results = []

        async def run_batch(batch_count: int):
            batch_results = []
            for _ in range(batch_count):
                result = self.run_round()
                batch_results.append(result)
            return batch_results

        batches = (count + batch_size - 1) // batch_size
        for _ in range(batches):
            batch_count = min(batch_size, count - len(results))
            if batch_count <= 0:
                break
            batch = await run_batch(batch_count)
            results.extend(batch)

        return results

    def get_statistics(self) -> Dict[str, Any]:
        """Get arena statistics."""
        total_wins = sum(self._win_counts.values())

        win_rates = {}
        for config in self.agent_configs:
            wins = self._win_counts.get(config.agent_id, 0)
            win_rates[config.agent_id] = wins

        return {
            "rounds_completed": self.rounds_completed,
            "win_rates": win_rates,
            "total_budget_consumed": self._total_budget_consumed,
            "agents": [
                {
                    "id": c.agent_id,
                    "strategy": c.strategy_type.value,
                    "budget": c.budget,
                }
                for c in self.agent_configs
            ],
        }

    def export_results(self) -> Dict[str, Any]:
        """Export all results for analysis."""
        return {
            "rounds": [r.to_dict() for r in self.results],
            "statistics": self.get_statistics(),
        }
```

```python
"""
Strategy Learner - Analyze negotiation outcomes to derive optimal strategies.

Phase 48.5: Tectonic Agent Academy

Learns from simulation results to produce StrategyProfile parameters
that maximize win rate and budget efficiency.
"""

import json
import logging
import math
from dataclasses import dataclass, field
from pathlib import Path
from typing import Dict, List, Any, Optional, Tuple
from statistics import mean, stdev

from systems.tectonic.arena import SimulationArena, SimulationResult
from systems.tectonic.strategy import StrategyProfile, StrategyType

logger = logging.getLogger("tectonic.learner")


@dataclass
class LearningConfig:
    """Configuration for strategy learning."""

    learning_rate: float = 0.1
    min_samples: int = 100
    distance_bins: int = 10
    bid_bins: int = 10


class StrategyLearner:
    """
    Analyzes negotiation outcomes to derive optimal strategies.

    Collects samples from arena simulations, analyzes patterns,
    and produces StrategyProfile parameters optimized for win rate.
    """

    def __init__(self, config: Optional[LearningConfig] = None):
        """Initialize the learner."""
        self.config = config or LearningConfig()
        self.samples: List[SimulationResult] = []
        self._distance_wins: Dict[int, int] = {}
        self._bid_wins: Dict[int, int] = {}
        self._total_by_distance: Dict[int, int] = {}
        self._total_by_bid: Dict[int, int] = {}

    @property
    def samples_collected(self) -> int:
        """Get number of samples collected."""
        return len(self.samples)

    def add_sample(self, result: SimulationResult) -> None:
        """Add a simulation result as a learning sample."""
        self.samples.append(result)

        # Bin by distance
        distance = math.sqrt(result.tile[0]**2 + result.tile[1]**2)
        dist_bin = int(distance / 10) * 10

        self._total_by_distance[dist_bin] = self._total_by_distance.get(dist_bin, 0) + 1
        self._distance_wins[dist_bin] = self._distance_wins.get(dist_bin, 0) + 1

        # Bin by bid
        bid_bin = int(result.winning_bid / 5) * 5

        self._total_by_bid[bid_bin] = self._total_by_bid.get(bid_bin, 0) + 1
        self._bid_wins[bid_bin] = self._bid_wins.get(bid_bin, 0) + 1

    def learn_from_arena(self, arena: SimulationArena) -> None:
        """Learn from all results in an arena."""
        for result in arena.results:
            self.add_sample(result)

    def derive_strategy(self) -> StrategyProfile:
        """
        Derive a learned strategy profile from collected samples.

        Returns:
            StrategyProfile with learned parameters
        """
        if len(self.samples) < 10:
            logger.warning("Not enough samples for learning, returning default")
            return StrategyProfile()

        # Analyze winning patterns
        winning_bids = [s.winning_bid for s in self.samples]
        avg_winning_bid = mean(winning_bids) if winning_bids else 10.0

        # Calculate average distance of winning tiles
        winning_distances = [
            math.sqrt(s.tile[0]**2 + s.tile[1]**2)
            for s in self.samples
        ]
        avg_distance = mean(winning_distances) if winning_distances else 50.0

        # Derive parameters
        # Base bid fraction: average winning bid / typical budget (100)
        base_fraction = min(0.3, max(0.05, avg_winning_bid / 100.0))

        # Distance weight: higher if far tiles win less
        near_wins = sum(1 for d in winning_distances if d < 50)
        far_wins = sum(1 for d in winning_distances if d >= 50)
        total = near_wins + far_wins

        if total > 0:
            near_rate = near_wins / total
            distance_weight = 0.1 if near_rate > 0.6 else 0.05
        else:
            distance_weight = 0.05

        # Aggression: based on bid variance
        if len(winning_bids) > 1:
            bid_variance = stdev(winning_bids)
            aggression = min(0.9, max(0.1, bid_variance / 20.0))
        else:
            aggression = 0.5

        return StrategyProfile(
            base_bid_fraction=base_fraction,
            distance_weight=distance_weight,
            aggression_level=aggression,
        )

    def analyze_distance_correlation(self) -> Dict[str, Any]:
        """Analyze correlation between distance and win rate."""
        correlations = []

        for dist_bin in sorted(self._total_by_distance.keys()):
            total = self._total_by_distance[dist_bin]
            wins = self._distance_wins.get(dist_bin, 0)
            rate = wins / total if total > 0 else 0.0
            correlations.append((dist_bin, rate))

        # Calculate simple correlation
        if len(correlations) > 1:
            distances = [c[0] for c in correlations]
            rates = [c[1] for c in correlations]
            mean_d = mean(distances)
            mean_r = mean(rates)

            numerator = sum((d - mean_d) * (r - mean_r) for d, r in correlations)
            denom_d = math.sqrt(sum((d - mean_d)**2 for d in distances))
            denom_r = math.sqrt(sum((r - mean_r)**2 for r in rates))

            if denom_d > 0 and denom_r > 0:
                correlation = numerator / (denom_d * denom_r)
            else:
                correlation = 0.0
        else:
            correlation = 0.0

        return {
            "correlation": correlation,
            "bins": correlations,
        }

    def analyze_bid_efficiency(self) -> Dict[str, Any]:
        """Analyze bid efficiency to find optimal bid range."""
        if not self._bid_wins:
            return {"optimal_bid_range": (5, 15)}

        # Find bid bins with highest win rates
        rates = []
        for bid_bin in sorted(self._bid_wins.keys()):
            wins = self._bid_wins[bid_bin]
            total = self._total_by_bid.get(bid_bin, 1)
            rate = wins / total if total > 0 else 0.0
            rates.append((bid_bin, rate))

        # Sort by rate
        rates.sort(key=lambda x: x[1], reverse=True)

        # Optimal range is around top bins
        if rates:
            top_bin = rates[0][0]
            optimal_range = (max(0, top_bin - 5), top_bin + 10)
        else:
            optimal_range = (5, 15)

        return {
            "optimal_bid_range": optimal_range,
            "bid_rates": rates[:5],
        }

    def export_learned_params(self) -> Dict[str, Any]:
        """Export learned parameters as dictionary."""
        profile = self.derive_strategy()
        return profile.to_dict()

    def save_profile(self, profile: StrategyProfile, path: str) -> None:
        """Save a profile to JSON file."""
        with open(path, 'w') as f:
            json.dump(profile.to_dict(), f, indent=2)
        logger.info(f"Saved strategy profile to {path}")

    def load_profile(self, path: str) -> StrategyProfile:
        """Load a profile from JSON file."""
        with open(path, 'r') as f:
            data = json.load(f)
        return StrategyProfile.from_dict(data)
```
